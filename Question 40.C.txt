#include <stdio.h>

int main() {
    int num, choice;

    printf("Enter a number: ");
    scanf("%d", &num);

    printf("\nChoose an option:\n");
    printf("1. Check if Buzz number\n");
    printf("2. Check if Even or Odd\n");
    printf("3. Check if Positive or Negative\n");
    printf("Enter your choice (1-3): ");
    scanf("%d", &choice);

    switch(choice) {
        case 1:
            if (num % 10 == 7 || num % 7 == 0)
                printf("Buzz number.\n");
            else
                printf("Not a Buzz number.\n");
            break;

        case 2:
            if (num % 2 == 0)
                printf("Even number.\n");
            else
                printf("Odd number.\n");
            break;

        case 3:
            if (num > 0)
                printf("Positive number.\n");
            else if (num < 0)
                printf("Negative number.\n");
            else
                printf("Number is zero.\n");
            break;

        default:
printf("nvalid choice.\n");
    }
    return 0;
